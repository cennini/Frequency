// Aumenta la risoluzione per le note basse analyser.fftSize = 8192; const bufferLength = analyser.frequencyBinCount; const dataArray = new Float32Array(bufferLength); // Usiamo Float per precisione function update() { // Usiamo getFloatFrequencyData per avere i decibel reali analyser.getFloatFrequencyData(dataArray); let maxDb = -Infinity; let maxIndex = -1; // Cerchiamo il picco solo nel range della chitarra (80Hz - 1000Hz) // Questo evita di prendere i 500Hz se non sono la nota base for (let i = 0; i < bufferLength; i++) { let freq = i * audioCtx.sampleRate / analyser.fftSize; if (freq > 70 && freq < 1200) { // Range chitarra if (dataArray[i] > maxDb) { maxDb = dataArray[i]; maxIndex = i; } } } if (maxDb > -60) { // Soglia di rumore: ignora se troppo piano const finalFreq = maxIndex * audioCtx.sampleRate / analyser.fftSize; freqDisplay.innerText = finalFreq.toFixed(1); } requestAnimationFrame(update); }
